Simple API Integration

Get started with just a few lines of code

// Initialize Open Banking connection const bankConnection = await monzo.openBanking.connect({ bank: 'barclays', account_id: 'user_123', redirect_uri: 'https://yourapp.com/callback' }); // Get real-time account balance const balance = await monzo.openBanking.getBalance( bankConnection.id ); console.log(balance.available); // ¬£2,847.50 // Retrieve 90 days of transactions const transactions = await monzo.openBanking.getTransactions({ connection_id: bankConnection.id, since: '2024-09-12', limit: 100 }); // Subscribe to real-time transaction updates await monzo.webhooks.subscribe({ event: 'transaction.created', connection_id: bankConnection.id, url: 'https://yourapp.com/webhooks' });
